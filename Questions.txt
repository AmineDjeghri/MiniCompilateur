EXEMPLES :

- Structure programme simple
ALGORITHME algo
VAR entier a;
reel b;
chaine c;
DEBUT
FIN


- Affectation de variable et remplissage des tableaux.
ALGORITHME algo
VAR entier a;
reel b;
chaine c;
DEBUT
 a<--5;
FIN



- Affectation mauvais type.
ALGORITHME algo
VAR entier a;
reel b;
chaine c;
DEBUT
 a<--5;
 a<--5.5;
 c<--2;
FIN

- Affectation ENTIER à un REEL.
ALGORITHME algo
VAR entier a;
reel b;
chaine c;
DEBUT
 b<--5;
FIN

- Affectation CHAINE.
ALGORITHME algo
VAR entier a;
reel b;
chaine c;
DEBUT
 c<--"bonjour";
FIN

- Quadruplés de calculs complexe : a+b+c;
ALGORITHME algo
VAR entier a;
reel b;
chaine c;
DEBUT
 a<--a+b+5;
FIN



- Jump de Faire SI
ALGORITHME algo
VAR entier a;
reel b;
chaine c;
DEBUT
Faire a<--5; SI (b<2)
FIN



- Dépassement taille tableau.
ALGORITHME algo
VAR a[3]:entier;
reel b;
chaine c;
DEBUT
 a[2]<--5;
 a[3]<--2;
FIN
